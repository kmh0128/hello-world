문제풀이
===


class Solution {
    public int[] solution(int[] numbers, int num1, int num2) {
        int[] answer = new int[num2 - num1 + 1];//배열설정 num2까지 수를 빼는것이므로 +1 

        for(int i = num1, j = 0; i < num2+1; i++) {//for문 
            answer[j] = numbers[i];//배열에 추가해줄때마다 카운트 해준다.
            j++;
        }

        return answer;
    }
}
