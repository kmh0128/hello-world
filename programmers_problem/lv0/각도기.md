class Solution {
    public int solution(int angle) {
        int answer = 0;

        if (angle > 0 && angle < 90)//90 미만이고 0보다 큰경우 1
            answer = 1;
        else if (angle == 90)//90일경우 0
            answer = 2;
        else if (angle > 90 && angle < 180)// 90보다 크고 or 180 미만일 경우 3
            answer = 3;
        else
            answer = 4; // 180보다 크면 4

        return answer;
    }
}
